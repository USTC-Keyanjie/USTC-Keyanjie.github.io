Given an input string (s) and a pattern (p), implement regular expression matching with support for '.' and '*'.

'.' Matches any single character.
'*' Matches zero or more of the preceding element.

The matching should cover the entire input string (not partial).

Note:

  • s could be empty and contains only lowercase letters a-z.
  • p could be empty and contains only lowercase letters a-z, and characters like . or *.

#include <iostream>
#include <algorithm>
using namespace std;

class Solution {
public:

    bool isMatch(string s, string p) {
        s = "0" + s;
        p = "0" + p;

        bool dp[p.length()][s.length()];

        dp[0][0] = true;
        for (int j = 1; j < s.length(); j++) {
            dp[0][j] = false;
        }
        for (int i = 1; i < p.length(); i++) {
            dp[i][0] = false;
        }

        for (int i = 2; i < p.length() && p[i] == '*'; i += 2) {
            dp[i][0] = true;
        }

        // æœ€åä¸€è¡Œå•ç‹¬è€ƒè™‘
        for(int i = 1; i < p.length() - 1;) {
            // ä¸‹ä¸€ä¸ªä¸æ˜¯'*'
            if (p[i + 1] != '*') {
                for(int j = 1; j < s.length(); j++) {
                    dp[i][j] = (p[i] == s[j] || p[i] == '.') ? dp[i-1][j-1] : false;
                }
                i++;
            } else {
                for(int j = 1; j < s.length(); j++) {
                    dp[i+1][j] = (p[i] == s[j] || p[i] == '.') ? (dp[i+1][j-1] || dp[i-1][j]) : dp[i-1][j];
                }
                i += 2;
            }
        }

        if (p[p.length() - 1] != '*') {
            for (int j = 1; j < s.length(); j++) {
                dp[p.length() - 1][j] = (p[p.length() - 1] == s[j] || p[p.length() - 1] == '.') ? dp[p.length() - 2][j-1] : false;
            }
        }
        cout << endl;

        return dp[p.length() - 1][s.length() - 1];
    }
};
